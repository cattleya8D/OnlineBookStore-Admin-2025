-- 1. 用户表（管理员/运营/仓管/客服）
CREATE TABLE Users (
    user_id BIGINT PRIMARY KEY AUTO_INCREMENT,
    username VARCHAR(50) UNIQUE NOT NULL,
    password_hash VARCHAR(255) NOT NULL,          -- 实际应哈希存储
    role ENUM('admin', 'operator', 'warehouse', 'support') NOT NULL,
    real_name VARCHAR(50),
    phone VARCHAR(20),
    email VARCHAR(100),
    create_time DATETIME DEFAULT CURRENT_TIMESTAMP,
    INDEX idx_role (role)
) COMMENT '系统用户表（权限角色区分）';

-- 2. 图书分类表
CREATE TABLE BookCategories (
    category_id BIGINT PRIMARY KEY AUTO_INCREMENT,
    category_name VARCHAR(100) NOT NULL,
    parent_id BIGINT DEFAULT 0,                    -- 支持二级分类
    description TEXT,
    create_time DATETIME DEFAULT CURRENT_TIMESTAMP,
    UNIQUE INDEX uk_category_name (category_name)
) COMMENT '图书分类表';

-- 3. 图书表
CREATE TABLE Books (
    book_id BIGINT PRIMARY KEY AUTO_INCREMENT,
    isbn VARCHAR(20) UNIQUE NOT NULL,
    title VARCHAR(200) NOT NULL,
    author VARCHAR(100),
    publisher VARCHAR(100),
    publish_year INT,
    price DECIMAL(10,2) NOT NULL,
    stock INT NOT NULL DEFAULT 0,
    category_id BIGINT,
    description TEXT,
    cover_url VARCHAR(255),
    create_time DATETIME DEFAULT CURRENT_TIMESTAMP,
    update_time DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    FOREIGN KEY (category_id) REFERENCES BookCategories(category_id),
    INDEX idx_title (title),
    INDEX idx_isbn (isbn),
    INDEX idx_category (category_id)
) COMMENT '图书信息表（库存同步关键表）';

-- 4. 客户表
CREATE TABLE Customers (
    customer_id BIGINT PRIMARY KEY AUTO_INCREMENT,
    name VARCHAR(50) NOT NULL,
    phone VARCHAR(20) UNIQUE,
    address VARCHAR(255),
    email VARCHAR(100),
    register_time DATETIME DEFAULT CURRENT_TIMESTAMP
) COMMENT '客户信息表';

-- 5. 订单主表
CREATE TABLE Orders (
    order_id BIGINT PRIMARY KEY AUTO_INCREMENT,
    customer_id BIGINT,
    user_id BIGINT,                               -- 操作员/客服
    order_date DATETIME DEFAULT CURRENT_TIMESTAMP,
    total_amount DECIMAL(12,2) NOT NULL,
    status ENUM('待付款','已付款','已发货','已完成','已取消') NOT NULL DEFAULT '待付款',
    payment_method VARCHAR(50),
    notes TEXT,
    FOREIGN KEY (customer_id) REFERENCES Customers(customer_id),
    FOREIGN KEY (user_id) REFERENCES Users(user_id),
    INDEX idx_status (status),
    INDEX idx_customer (customer_id),
    INDEX idx_order_date (order_date)
) COMMENT '订单主表（状态机核心）';

-- 6. 订单明细表（主从表设计）
CREATE TABLE OrderDetails (
    detail_id BIGINT PRIMARY KEY AUTO_INCREMENT,
    order_id BIGINT NOT NULL,
    book_id BIGINT NOT NULL,
    quantity INT NOT NULL,
    unit_price DECIMAL(10,2) NOT NULL,
    subtotal DECIMAL(12,2) NOT NULL,
    FOREIGN KEY (order_id) REFERENCES Orders(order_id) ON DELETE CASCADE,
    FOREIGN KEY (book_id) REFERENCES Books(book_id),
    INDEX idx_order (order_id)
) COMMENT '订单明细表（多对多关联）';

-- 7. 物流记录表
CREATE TABLE ShippingRecords (
    shipping_id BIGINT PRIMARY KEY AUTO_INCREMENT,
    order_id BIGINT NOT NULL UNIQUE,
    tracking_number VARCHAR(100),
    ship_date DATETIME,
    status ENUM('待发货','已揽件','运输中','已签收') DEFAULT '待发货',
    company VARCHAR(100),
    notes TEXT,
    FOREIGN KEY (order_id) REFERENCES Orders(order_id) ON DELETE CASCADE,
    INDEX idx_order_shipping (order_id)
) COMMENT '物流发货跟踪表';

-- 8. 评价表
CREATE TABLE Reviews (
    review_id BIGINT PRIMARY KEY AUTO_INCREMENT,
    order_id BIGINT,
    book_id BIGINT NOT NULL,
    customer_id BIGINT NOT NULL,
    rating TINYINT NOT NULL CHECK (rating BETWEEN 1 AND 5),
    comment_text TEXT,
    review_date DATETIME DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (order_id) REFERENCES Orders(order_id),
    FOREIGN KEY (book_id) REFERENCES Books(book_id),
    FOREIGN KEY (customer_id) REFERENCES Customers(customer_id),
    INDEX idx_book (book_id),
    INDEX idx_customer (customer_id)
) COMMENT '客户评价表（收货后评价）';
